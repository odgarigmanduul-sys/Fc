Shader "Mongolingo/OmniFX" {
    Properties {
        _MainTex ("Screen", 2D) = "white" {}
        _WagerLevel ("Wager (1:Fire, 2:Lava, 3:Galaxy)", Range(0,3)) = 0
    }
    SubShader {
        Tags { "RenderType"="Transparent" "Queue"="Transparent" }
        Blend SrcAlpha OneMinusSrcAlpha
        Cull Off ZWrite Off

        Pass {
            HLSLPROGRAM
            #pragma vertex vert
            #pragma fragment frag
            #include "UnityCG.cginc"

            struct appdata {
                float4 vertex : POSITION;
                float2 uv : TEXCOORD0;
            };

            struct v2f {
                float2 uv : TEXCOORD0;
                float4 vertex : SV_POSITION;
            };

            sampler2D _MainTex;
            float _WagerLevel;

            v2f vert (appdata v) {
                v2f o;
                o.vertex = UnityObjectToClipPos(v.vertex);
                o.uv = v.uv;
                return o;
            }

            float4 frag (v2f i) : SV_Target {
                float2 uv = i.uv;
                float4 col = tex2D(_MainTex, uv);

                // LAVA MODE (Distortion + Volcanic Glow)
                if (_WagerLevel >= 2.0) {
                    uv.y += sin(uv.x * 15.0 + _Time.y * 10.0) * 0.03;
                    col = tex2D(_MainTex, uv) + float4(1.0, 0.3, 0, 1) * 0.6;
                }

                // GALAXY MODE (Fractal Singularity)
                if (_WagerLevel >= 3.0) {
                    float star = frac(sin(dot(uv, float2(12.98, 78.23))) * 43758.54);
                    if(star > 0.99) col += float4(0.5, 0, 1, 1) * 3.0;
                }

                return col;
            }
            ENDHLSL
        }
    }
}
